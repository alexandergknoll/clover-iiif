# PartOf

A containing resource that includes the resource that has the partOf property.

import { Primitives as IIIF } from "@samvera/clover-iiif";

<IIIF.PartOf
  partOf={[
    {
      id: "https://digital.lib.utk.edu/assemble/collection/collections/rftaart",
      type: "Collection",
      label: { none: ["Parent Collection"] },
    },
    {
      id: "https://digital.lib.utk.edu/assemble/collection/collections/rftacuratedart",
      type: "Collection",
    },
  ]}
/>;
